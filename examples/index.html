Code Examples

Vanilla JavaScript

// Create a basic dialog var dialog = new FlowDialog(document.getElementById('myDialog'), { width: 600, closeOnEscape: true, closeOnOverlayClick: true }); // Open the dialog dialog.open(); // Flow to next step dialog.flow('next'); // Listen for events dialog.on('onOpen', function() { console.log('Dialog opened!'); });

jQuery

// Initialize and open $('#myDialog').flowdialog({ width: 600, closeOnEscape: true }).flowdialog('open'); // Flow navigation $('#myDialog').flowdialog('flow', 'next'); // Event handling $('#myDialog').on('flowdialog_onOpen', function() { console.log('Dialog opened!'); });
